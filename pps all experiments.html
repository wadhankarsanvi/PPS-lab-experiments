def matmult(A, B):
	rows_A = len(A)
	cols_A = len(A[0])

	rows_B = len(B)
	cols_B = len(B[0])

	if cols_A != rows_B:
		print("Cannot multiply the two matrices. Incorrect dimensions.")
		return None

	result = []
	for i in range(rows_A):
		row = []
		for j in range(cols_B):
			row.append(0)
		result.append(row)

	for i in range(rows_A):
		for j in range(cols_B):
			for k in range(cols_A):
				result[i][j] += A[i][k] * B[k][j]
	return result
